## ğŸ“Š Database Schema

```sql
-- agents table (Legacy - for manual agents)
CREATE TABLE agents (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    label TEXT,
    model TEXT NOT NULL,
    task TEXT,
    status TEXT NOT NULL,
    progress INTEGER DEFAULT 0,
    startTime INTEGER,
    endTime INTEGER,
    tokensIn INTEGER DEFAULT 0,
    tokensOut INTEGER DEFAULT 0,
    createdAt INTEGER NOT NULL,
    updatedAt INTEGER NOT NULL
);

-- sessions table (NEW - Clawdbot sessions)
CREATE TABLE sessions (
    id TEXT PRIMARY KEY,
    sessionKey TEXT NOT NULL,
    kind TEXT NOT NULL CHECK(kind IN ('direct', 'group')),
    model TEXT,
    contextTokens INTEGER,
    inputTokens INTEGER DEFAULT 0,
    outputTokens INTEGER DEFAULT 0,
    totalTokens INTEGER DEFAULT 0,
    startTime INTEGER,
    lastActivity INTEGER,
    endTime INTEGER,
    status TEXT DEFAULT 'running',
    aborted BOOLEAN DEFAULT 0,
    createdAt INTEGER NOT NULL,
    updatedAt INTEGER NOT NULL
);

-- logs table (shared by both agents and sessions)
CREATE TABLE logs (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    agentId TEXT,
    sessionId TEXT,
    message TEXT NOT NULL,
    level TEXT NOT NULL,
    timestamp INTEGER NOT NULL,
    FOREIGN KEY (agentId) REFERENCES agents(id) ON DELETE CASCADE,
    FOREIGN KEY (sessionId) REFERENCES sessions(id) ON DELETE CASCADE
);

-- session_messages table (NEW - conversation history)
CREATE TABLE session_messages (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    sessionId TEXT NOT NULL,
    role TEXT NOT NULL,
    content TEXT,
    tokens INTEGER,
    timestamp INTEGER NOT NULL,
    FOREIGN KEY (sessionId) REFERENCES sessions(id) ON DELETE CASCADE
);

-- Indexes
CREATE INDEX idx_sessions_status ON sessions(status);
CREATE INDEX idx_sessions_lastActivity ON sessions(lastActivity);
CREATE INDEX idx_logs_session ON logs(sessionId, timestamp);
CREATE INDEX idx_messages_session ON session_messages(sessionId, timestamp);
```
